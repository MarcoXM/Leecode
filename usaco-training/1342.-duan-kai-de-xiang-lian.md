# 1342. 断开的项链

```python


def get(c):
    if c == "b":
        return 1
    return 2

N= int(input())
s = [x for x in  input()]

s += s
ans = 0
for i in range(N):
    l = i
    r = i + N - 1
    cnt = left = right = 0
    while l <= r and (s[l] == "w" or left|get(s[l]) != 3):
        if s[l] != "w" :
            left |= get(s[l])
        cnt += 1
        l += 1
            
    while l <= r and (s[r] == "w" or right|get(s[r]) != 3):
        if s[r] != "w" :
            right |= get(s[r])
        cnt += 1
        r -= 1
            
    ans = max(ans, cnt)
    
print(ans)
```
