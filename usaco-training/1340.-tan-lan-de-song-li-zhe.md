#include <iostream>
#include <unordered_map>
#include <cstring>
#include <stack>
using namespace std;

unordered_map<string,int> nameid;
string order[10];
int amount, cnt;
string name;
int main() {
    int N;
    cin >> N ;
    
    for (int i = 0; i < N ; i++){
        cin >> name;
        nameid[name] = 0;
        order[i] = name;
    }
    for (int i = 0 ; i < N; i++){
        cin >> name;
        cin >> amount >>cnt;
        if (!amount) continue;

        nameid[name] -=  amount/cnt * cnt;
        for (int j=0 ; j < cnt;j ++){
            cin >> name;
            nameid[name] += amount /cnt ;
        }
    }
    for(int i=0;i<N;i++) 
        cout << order[i] << " "<< nameid[order[i]]<< endl;
    return 0;
    
    
    
    
    
}

N = int(input())
dit = {}
ans = []
for _ in range(N):
    name = input()
    dit[name] = 0
    ans.append(name)
    
for _ in range(N):
    payer = input()
    data = input()
    amount, n = map(int,data.split())
    if not n:continue
    per = amount //n
    for _ in range(n):
        dit[input()] += per
        
    dit[payer] -= per * n
for name in ans:
    print(name, dit[name])
