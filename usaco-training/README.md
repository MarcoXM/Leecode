//Go sort slices of slices, 要descend 就修改less

type Element [][]int

func (p Element) Swap(i, j int)      { p[i], p[j] = p[j], p[i] }
func (p Element) Len() int           { return len(p) }
func (p Element) Less(i, j int) bool {
    if p[i][0] != p[j][0] {
        return p[i][0] < p[j][0] 
    } else {
        return p[i][1] < p[j][1]
    }
}
