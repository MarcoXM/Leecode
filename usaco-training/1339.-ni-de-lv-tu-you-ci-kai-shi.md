#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int helper(string s){
    int ans = 1;
    for (char c:s){
        ans *= (c - 'A' + 1) % 47;
    }
    return ans% 47;
}
string A,B;

int main() {
    cin >> A >> B;
    if (helper(A) == helper(B)){
        cout << "GO" << endl;
    }
     else {
        cout << "STAY" << endl;
    }
    return 0;
}
