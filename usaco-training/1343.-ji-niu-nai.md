# 1343. 挤牛奶

```go
package main
import (
go
    "sort"
    )


func min(a, b int) int {
    if a < b {
        return a
    } else {
        return b
    }
}

func max(a, b int) int {
    if a < b {
        return b
    } else {
        return a
    }
}

type Element [][]int

func (p Element) Swap(i, j int)      { p[i], p[j] = p[j], p[i] }
func (p Element) Len() int           { return len(p) }
func (p Element) Less(i, j int) bool {
    if p[i][0] != p[j][0] {
        return p[i][0] < p[j][0] 
    } else {
        return p[i][1] < p[j][1]
    }
}


func main() {
    var N,s,e int
    fmt.Scanf("%d", &N)
    arr := make([][]int,0)
    for i:= 0 ; i < N ; i ++{
        fmt.Scanf("%d %d", &s, &e)
        arr = append(arr,[]int{s,e})
    }
    sort.Sort(Element(arr))
    var ans1, ans2, l , r int
    l,r = arr[0][0], arr[0][1]
    for i:= 1 ; i < len(arr); i++{
        if arr[i][0] > r{
            ans1 = max( ans1, arr[i][0] - r)
            l = arr[i][0]
            r = arr[i][1]
        }else {
            r = max(r, arr[i][1])
            ans2 = max(ans2, r - l)
        }
    }
    ans2 = max(ans2, r - l)
    fmt.Printf("%d %d",ans2, ans1)
}
```
