# 1782. Count Pairs Of Nodes

```python
class Solution:
    def countPairs(self, n: int, edges: List[List[int]], queries: List[int]) -> List[int]:
        ans = []
        d = [0] *  (n + 1)
        cnt = collections.defaultdict(int)
        for e in edges:
            a = e[0]
            b = e[1]
            if a > b :
                a , b = b ,a 
            cnt[ a * 100000 + b] += 1
            d[a] += 1
            d[b] += 1

        ds = sorted(d[1:])
        for q in queries:
            s1 = s2 = s3 = 0
            for k, v in cnt.items():
                a = k //100000
                b = k % 100000
                if d[a] + d[b] - v > q: s1 += 1
                if d[a] + d[b] > q : s2 += 1

            i = n - 1
            j = 0
            while i  > j :
                while i > j and ds[i] + ds[j] <= q:
                    j += 1
                if j < i and ds[i] + ds[j] > q:
                    s3 += (i - j)
                i -= 1
            ans.append(s1 + s3 - s2)
        return ans
```
