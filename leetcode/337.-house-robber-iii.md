#         self.left = None
#         self.right = None

class Solution:
    def rob(self, root: TreeNode) -> int:
        ans = self.dfs(root)
        return max(ans)


    def dfs(self, root):
        if not root:
            return [0,0]
        money  = [0, 0]

        left = self.dfs(root.left)
        right = self.dfs(root.right)

        money[0] = max(left[0], left[1]) + max(right[0], right[1]);
        money[1] = left[0] + right[0] + root.val;
