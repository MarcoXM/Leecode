class Solution:
    def hammingDistance(self, x: int, y: int) -> int:
        ans = 0
        while x or y:
            # print(x,y)
            if x%2 != y%2:
                ans+=1
                
            x = x >> 1
            y = y >> 1
        return ans

class Solution:
    def hammingDistance(self, x: int, y: int) -> int:
        xor = x ^ y
        res = 0
        while xor:
            res += xor & 1
            xor >>= 1
        return res
