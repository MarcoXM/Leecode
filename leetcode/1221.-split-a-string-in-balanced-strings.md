# 1221. Split a String in Balanced Strings

```python
class Solution:
    def balancedStringSplit(self, s: str) -> int:
        if not s:
            return 0 # corner case
        ans = 0
        from collections import defaultdict
        dit = defaultdicï½”(int)
        for i in range(len(s)):
            if s[i] == "R":
                dit["R"] +=1
            elif s[i] =="L":
                dit["L"] +=1
            if dit["R"] == dit["L"]:
                ans += 1
                dit["R"] = 0
                dit["L"] = 0
        return ans
```
