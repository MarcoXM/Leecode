# 1630. Arithmetic Subarrays

```python
class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        ans = []
        for i, j in zip(l,r):
            if self.check(nums,i,j):
                ans.append(True)
            else:
                ans.append(False)
                
        return ans
            
            
            
    
    def check(self,A,i,j):
        nums = A[i:j+1]
        nums.sort()
        c = nums[1] - nums[0]
        for i in range(2,len(nums)):
            if nums[i] - nums[i-1] !=c:
                return False
            
        return True
```
