# 621. Task Scheduler

```python
class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        dit = collections.defaultdict(int)
        for t in tasks:
            dit[t] += 1
            
        h = [(-v, k)for k,v in dit.items()]
        heapq.heapify(h)
        ans = 0
        while h :
            idx = 0
            temp = []
            while idx <= n:
                if h:
                    count, word = heapq.heappop(h)
                    ## 0 我们就不放回去了
                    if count != -1:
                        temp.append((count + 1,word))
                ans += 1
                if not h and not temp:
                    return ans
                idx += 1
            for t in temp:
                heapq.heappush(h,t)
                
        return ans
```
