# 973. K Closest Points to Origin

```python

class Solution:
    def kClosest(self, points: List[List[int]], K: int) -> List[List[int]]:
        ### top k 还是heap吧
        ## 转化
        p = list(map(lambda x: ((x[0]**2 + x[1]**2),x),points))
        heapq.heapify(p)
        ans = []
        while K > 0:
            ans.append(heapq.heappop(p)[1])
            K -= 1
            
        return ans
```
